52  /**
53   * This is a helper class that provides a way to easily apply an Schematron
54   * resource on an XML resource.
55   *
56   * @author Philip Helger
57   */
58  @Immutable
59  public final class SchematronHelper

62  
63    @PresentForCodeCoverage
64    @SuppressWarnings ("unused")
65    private static final SchematronHelper s_aInstance = new SchematronHelper ();

67    private SchematronHelper ()
68    {}

70    /**
71     * Apply the passed schematron on the passed XML resource using a custom error
72     * handler.
73     *
74     * @param aSchematron
75     *        The Schematron resource. May not be <code>null</code>.
76     * @param aXML
77     *        The XML resource. May not be <code>null</code>.
78     * @return <code>null</code> if either the Schematron or the XML could not be
79     *         read.
80     * @throws IllegalStateException
81     *         if the processing throws an unexpected exception.
82     */
83    @Nullable
84    public static SchematronOutputType applySchematron (@Nonnull final ISchematronResource aSchematron,
85                                                        @Nonnull final IReadableResource aXML)
86    {
87      ValueEnforcer.notNull (aSchematron, "SchematronResource");
88      ValueEnforcer.notNull (aXML, "XMLSource");
89  
90      try
91      {
92        // Apply Schematron on XML
93        return aSchematron.applySchematronValidationToSVRL (aXML);
94      }
95      catch (final Exception ex)
96      {
97        throw new IllegalArgumentException ("Failed to apply Schematron " +
98                                            aSchematron.getID () +
99                                            " onto XML resource " +
100                                           aXML.getResourceID (), ex);
101     }
102   }

159   /**
160    * Convert a {@link SchematronOutputType} to an {@link IResourceErrorGroup}.
161    *
162    * @param aSchematronOutput
163    *        The result of Schematron validation
164    * @param sResourceName
165    *        The name of the resource that was validated (may be a file path
166    *        etc.)
167    * @return List non-<code>null</code> error list of {@link SVRLResourceError}
168    *         objects.
169    */
170   @Nonnull
171   public static IResourceErrorGroup convertToResourceErrorGroup (@Nonnull final SchematronOutputType aSchematronOutput,
172                                                                  @Nullable final String sResourceName)
173   {
174     ValueEnforcer.notNull (aSchematronOutput, "SchematronOutput");
175 
176     final ResourceErrorGroup ret = new ResourceErrorGroup ();
177     for (final SVRLFailedAssert aFailedAssert : SVRLUtils.getAllFailedAssertions (aSchematronOutput))
178       ret.addResourceError (aFailedAssert.getAsResourceError (sResourceName));
179     return ret;
180   }

182   @SuppressFBWarnings ("RCN_REDUNDANT_NULLCHECK_OF_NONNULL_VALUE")
183   private static void _recursiveResolveAllSchematronIncludes (@Nonnull final IMicroElement eRoot,
184                                                               @Nonnull final IReadableResource aResource,
185                                                               @Nullable final ISAXReaderSettings aSettings)
186   {
187     if (eRoot != null)
188     {
189       final DefaultSchematronIncludeResolver aIncludeResolver = new DefaultSchematronIncludeResolver (aResource);
190 
191       for (final IMicroElement aElement : eRoot.getAllChildElementsRecursive ())
192         if (CSchematron.NAMESPACE_SCHEMATRON.equals (aElement.getNamespaceURI ()) &&
193             aElement.getLocalName ().equals (CSchematronXML.ELEMENT_INCLUDE))
194         {
195           String sHref = aElement.getAttributeValue (CSchematronXML.ATTR_HREF);
196           try
197           {
198             final int nHashIndex = sHref.indexOf ('#');
199             String sAnchor = null;
200             if (nHashIndex >= 0)
201             {
202               sAnchor = sHref.substring (nHashIndex + 1);
203               sHref = sHref.substring (0, nHashIndex);
204             }
205 
206             final IReadableResource aIncludeRes = aIncludeResolver.getResolvedSchematronResource (sHref);
207             if (aIncludeRes == null)
208               throw new IllegalStateException ("Failed to resolve include '" + sHref + "'");
209 
210             if (s_aLogger.isDebugEnabled ())
211               s_aLogger.debug ("Resolved '" +
212                                sHref +
213                                "' relative to '" +
214                                aIncludeResolver.getBaseHref () +
215                                "' as '" +
216                                aIncludeRes.getPath () +
217                                "'");
218 
219             // Read XML to be included
220             final IMicroDocument aIncludedDoc = MicroReader.readMicroXML (aIncludeRes, aSettings);
221             if (aIncludedDoc == null)
222               throw new IllegalStateException ("Failed to parse include " + aIncludeRes);
223 
224             IMicroElement aIncludedContent;
225             if (sAnchor == null)
226             {
227               // no anchor present - include the whole document
228 
229               // Return the document element
230               aIncludedContent = aIncludedDoc.getDocumentElement ();
231             }
232             else
233             {
234               final String sFinalAnchor = sAnchor;
235               final Wrapper <IMicroElement> aMatch = new Wrapper <IMicroElement> ();
236               MicroWalker.walkNode (aIncludedDoc.getDocumentElement (),
237                                     new DefaultHierarchyWalkerCallback <IMicroNode> ()
238                                     {
239                                       @Override
240                                       public void onItemBeforeChildren (final IMicroNode aItem)
241                                       {
242                                         if (aItem.isElement ())
243                                         {
244                                           final IMicroElement aCurElement = (IMicroElement) aItem;
245                                           final String sID = aCurElement.getAttributeValue ("id");
246                                           if (sFinalAnchor.equals (sID))
247                                             aMatch.set (aCurElement);
248                                         }
249                                       }
250                                     });
251               aIncludedContent = aMatch.get ();
252               if (aIncludedContent == null)
253               {
254                 s_aLogger.error ("Failed to resolve an element with the ID '" +
255                                  sAnchor +
256                                  "' in " +
257                                  aIncludeRes +
258                                  "! Therefore including the whole document!");
259                 aIncludedContent = aIncludedDoc.getDocumentElement ();
260               }
261             }
262 
263             // Important to detach from parent!
264             aIncludedContent.detachFromParent ();
265 
266             // Recursive resolve includes
267             _recursiveResolveAllSchematronIncludes (aIncludedContent, aIncludeRes, aSettings);
268 
269             // Now replace "include" element with content in MicroDOM
270             aElement.getParent ().replaceChild (aElement, aIncludedContent);
271           }
272           catch (final IOException ex)
273           {
274             throw new IllegalStateException ("Failed to read include '" + sHref + "'", ex);
275           }
276         }
277     }
278   }

280   /**
281    * Resolve all Schematron includes of the passed resource.
282    *
283    * @param aResource
284    *        The Schematron resource to read. May not be <code>null</code>.
285    * @return <code>null</code> if the passed resource could not be read as XML
286    *         document
287    */
288   @Nullable
289   public static IMicroDocument getWithResolvedSchematronIncludes (@Nonnull final IReadableResource aResource)
290   {
291     return getWithResolvedSchematronIncludes (aResource, (ISAXReaderSettings) null);
292   }
293 
294   /**
295    * Resolve all Schematron includes of the passed resource.
296    *
297    * @param aResource
298    *        The Schematron resource to read. May not be <code>null</code>.
299    * @param aSettings
300    *        The SAX reader settings to be used. May be <code>null</code> to use
301    *        the default settings.
302    * @return <code>null</code> if the passed resource could not be read as XML
303    *         document
304    */
305   @Nullable
306   public static IMicroDocument getWithResolvedSchematronIncludes (@Nonnull final IReadableResource aResource,
307                                                                   @Nullable final ISAXReaderSettings aSettings)
308   {
309     final IMicroDocument aDoc = MicroReader.readMicroXML (aResource, aSettings);
310     if (aDoc != null)
311     {
312       // Resolve all Schematron includes
313       _recursiveResolveAllSchematronIncludes (aDoc.getDocumentElement (), aResource, aSettings);
314     }
315     return aDoc;
316   }
