17  package com.helger.schematron.pure.binding.xpath;
18  
19  import java.util.Map;
20  import java.util.TreeMap;
21  
22  import javax.annotation.Nonnull;
23  import javax.annotation.Nullable;
24  import javax.annotation.concurrent.NotThreadSafe;
25  
26  import com.helger.commons.ValueEnforcer;
27  import com.helger.commons.annotations.Nonempty;
28  import com.helger.commons.annotations.ReturnsMutableCopy;
29  import com.helger.commons.compare.ComparatorStringLongestFirst;
30  import com.helger.commons.state.EChange;
31  import com.helger.commons.string.StringHelper;
32  
33  /**
34   * This class manages all variables present in Schematron &lt;let&gt; elements.
35   *
36   * @author Philip Helger
37   */
38  @NotThreadSafe
39  public class PSXPathVariables
40  {
41    private final Map <String, String> m_aMap = new TreeMap <String, String> (new ComparatorStringLongestFirst ());
42  
43    /**
44     * Add a new variable.
45     *
46     * @param aEntry
47     *        The entry to be added - key is the variable name and value is the
48     *        variable value. May not be <code>null</code>.
49     * @return {@link EChange#UNCHANGED} if a variable with the same name is
50     *         already present. Never <code>null</code>.
51     */
52    @Nonnull
53    public EChange add (@Nonnull final Map.Entry <String, String> aEntry)
54    {
55      return add (aEntry.getKey (), aEntry.getValue ());
56    }
57  
58    /**
59     * Add a new variable.
60     *
61     * @param sName
62     *        The name of the variable. May neither be <code>null</code> nor
63     *        empty.
64     * @param sValue
65     *        The value of the variable. May neither be <code>null</code> nor
66     *        empty.
67     * @return {@link EChange#UNCHANGED} if a variable with the same name is
68     *         already present. Never <code>null</code>.
69     */
70    @Nonnull
71    public EChange add (@Nonnull @Nonempty final String sName, @Nonnull @Nonempty final String sValue)
72    {
73      ValueEnforcer.notEmpty (sName, "Name");
74      ValueEnforcer.notEmpty (sValue, "Value");
75  
76      // Prepend the "$" prefix to the variable name
77      final String sRealName = PSXPathQueryBinding.PARAM_VARIABLE_PREFIX + sName;
78      if (m_aMap.containsKey (sRealName))
79        return EChange.UNCHANGED;
80  
81      // Apply all existing variables to this variable value!
82      final String sRealValue = getAppliedReplacement (sValue);
83      m_aMap.put (sRealName, sRealValue);
84      return EChange.CHANGED;
85    }
86  
87    /**
88     * Perform the text replacement of all variables in the specified text.
89     *
90     * @param sText
91     *        The source text. May be <code>null</code>.
92     * @return The text with all values replaced. May be <code>null</code> if the
93     *         source text is <code>null</code>.
94     */
95    @Nullable
96    public String getAppliedReplacement (@Nullable final String sText)
97    {
98      return PSXPathQueryBinding.getWithParamTextsReplacedStatic (sText, m_aMap);
99    }
100 
101   /**
102    * Remove the variable with the specified name
103    *
104    * @param sVarName
105    *        The name of the variable to be removed. May be <code>null</code>.
106    * @return {@link EChange#CHANGED} if the variable was removed successfully.
107    *         Never <code>null</code>.
108    */
109   @Nonnull
110   public EChange remove (@Nullable final String sVarName)
111   {
112     if (StringHelper.hasText (sVarName))
113       if (m_aMap.remove (PSXPathQueryBinding.PARAM_VARIABLE_PREFIX + sVarName) == null)
114         return EChange.CHANGED;
115     return EChange.UNCHANGED;
116   }
117 
118   /**
119    * Remove all variables with the specified names
120    *
121    * @param aVars
122    *        A list of variable names to be removed. May be <code>null</code>.
123    * @return {@link EChange#CHANGED} if at least one variable was removed
124    *         successfully. Never <code>null</code>.
125    */
126   @Nonnull
127   public EChange removeAll (@Nullable final Iterable <String> aVars)
128   {
129     EChange eChange = EChange.UNCHANGED;
130     if (aVars != null)
131       for (final String sName : aVars)
132         eChange = eChange.or (remove (sName));
133     return eChange;
134   }
135 
136   @Nonnull
137   @ReturnsMutableCopy
138   public Map <String, String> getAll ()
139   {
140     final Map <String, String> ret = new TreeMap <String, String> (new ComparatorStringLongestFirst ());
141     ret.putAll (m_aMap);
142     return ret;
143   }
144 
145   public boolean contains (@Nullable final String sName)
146   {
147     if (StringHelper.hasNoText (sName))
148       return false;
149     return m_aMap.containsKey (sName);
150   }
151 
152   @Nullable
153   public String get (@Nullable final String sName)
154   {
155     if (StringHelper.hasNoText (sName))
156       return null;
157     return m_aMap.get (sName);
158   }
159 }
